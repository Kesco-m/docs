849  private void RenderTableBody(StringWriter w, DataView dv, ref decimal calcSum, ref decimal calcSumE)
850  {
851  DateTime dTime;
852  string _dTime = "";
853 
854  decimal sum = 0M;
855  decimal sumE = 0M;
856 
857  string title = GetTitleByColor();
858  for (int i = 0; i < dv.Count; i++)
859  {
860  w.Write("<tr");
861  if (i%2 != 0)
862  w.Write(" style='background:#efefef;'");
863  w.Write(">");
864  if (dv[i]["НачалоРазговора"] == null || dv[i]["НачалоРазговора"].ToString().Length == 0)
865  _dTime = "";
866  else
867  {
868  dTime = (DateTime) dv[i]["НачалоРазговора"];
869  _dTime = dTime.ToString("dd.MM.yy HH:mm:ss");
870  }
871 
872  w.Write("<td class='tdR' noWrap>{0}</td>", _dTime);
873  w.Write("<td class='tdC'>{0}</td>", dv[i]["Роуминг"]);
874  w.Write("<td class='tdC'>{0}</td>", dv[i]["Телефон"]);
875  w.Write("<td class='tdC'>{0}</td>", dv[i]["Тип"]);
876  w.Write("<td >{0}</td>", dv[i]["Услуга"]);
877  w.Write("<td >{0}</td>", dv[i]["Описание"]);
878  w.Write("<td class='tdR' noWrap>{0}</td>",
879  Convert.Second2TimeFormat(int.Parse(dv[i]["Секунд"].ToString())));
880  w.Write("<td class='tdR' noWrap>{0}</td>", dv[i]["Килобайт"]);
881  w.Write("<td class='tdR' noWrap style='{0}' {1}>", _qsParams["Color"], title);
882 
883 
884  sum = Convert.Round((decimal) dv[i]["Сумма"], _scale);
885  sumE = Convert.Round((decimal) dv[i]["СуммаСотрудника"], _scale);
886 
887  if (
888  System.Convert.ToInt32(_qsParams["NDSAllSumm"])
889  .Equals(System.Convert.ToInt32(TarifMobilPhone.СalculationNDS.OnCostOfEachCall)))
890  {
891  sum = Convert.Round(sum*(1 + _ndsStavka), _scale);
892  sumE = Convert.Round(sumE*(1 + _ndsStavka), _scale);
893  }
894 
895  calcSum += sum;
896  calcSumE += sumE;
897
