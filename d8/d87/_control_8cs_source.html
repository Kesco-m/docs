1012  public virtual bool DirectBind(ref object val, Type type, BindingDirection direction)
1013  {
1014  var changed = false;
1015  if (direction == BindingDirection.FromSource)
1016  {
1017  if (val == null)
1018  {
1019  Value = "";
1020  }
1021  else if (type == typeof (DateTime))
1022  {
1023  Value = ((DateTime) val).ToString("dd.MM.yyyy");
1024  }
1025  else if (type == typeof (DateTime?))
1026  {
1027  Value = ((DateTime?) val).Value.ToString("dd.MM.yyyy");
1028  }
1029  else if (type == typeof (bool))
1030  {
1031  Value = (bool) val ? "1" : "0";
1032  }
1033  else if (type == typeof (bool?))
1034  {
1035  Value = (bool) val ? "1" : "0";
1036  }
1037  else
1038  {
1039  Value = val.ToString();
1040  }
1041  }
1042  else
1043  {
1044  object newval = null;
1045  if (type == typeof (string))
1046  {
1047  newval = Value;
1048  }
1049  else if (type == typeof (int) || type == typeof (int?))

1054  }
1055  }

1061  }
1062  }

1066  }
