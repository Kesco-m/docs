411  private string SqlLoadUsls(string id)
412  {
413  return string.Format(@"
414 IF object_id('tempdb..#UslData') IS NOT NULL DROP TABLE #UslData
415 CREATE TABLE #UslData (
416  КодОказаннойУслуги int,
417  GuidОказаннойУслуги uniqueidentifier PRIMARY KEY,
418  КодДокумента int, Агент1 tinyint, Агент2 tinyint,
419  КодУчасткаОтправкиВагона int, КодОтправкиВагона int, ОтправкаВагона nvarchar(300), НомерВагона varchar(100),
420  КодРесурса int, РесурсРус nvarchar(300), РесурсЛат varchar(300), Количество float, КодЕдиницыИзмерения int, Коэффициент float,
421  ЦенаБезНДС money, КодСтавкиНДС int, СуммаНДС money, СуммаБезНДС money, Всего money, Порядок int, Изменил int, Изменено datetime,
422  Перевыставлена int)
423 
424 INSERT #UslData
425 SELECT
426  КодОказаннойУслуги, GuidОказаннойУслуги, КодДокумента, Агент1, Агент2,
427  КодУчасткаОтправкиВагона, NULL КодОтправкиВагона, '' ОтправкаВагона, '' НомерВагона,
428  КодРесурса, РесурсРус, РесурсЛат, Количество, КодЕдиницыИзмерения, Коэффициент,
429  ЦенаБезНДС, КодСтавкиНДС, СуммаНДС, СуммаБезНДС, Всего,
430 Порядок, Изменил, Изменено, 0 Перевыставлена
431 FROM vwОказанныеУслуги (nolock)
432 WHERE КодДокумента = {0}
433 
434 UPDATE #UslData
435 SET Перевыставлена = 1
436 FROM #UslData
437 WHERE EXISTS(SELECT * FROM НаборыУслуг (nolock) WHERE GuidОказаннойУслугиВходящей = #UslData.GuidОказаннойУслуги)
438 
439 UPDATE #UslData
440 SET НомерВагона = CONVERT(varchar, Участки.НомерВагона) + ISNULL(' [' + CONVERT(varchar, Участки.ДатаОтправления, 104) + ']',''),
441  КодОтправкиВагона = Участки.КодОтправкиВагона,
442  ОтправкаВагона = Отправки.ОтправкаВагона
443 FROM #UslData
444  INNER JOIN vwОтправкаВагоновУчастки Участки ON #UslData.КодУчасткаОтправкиВагона = Участки.КодУчасткаОтправкиВагона
445  INNER JOIN vwОтправкаВагонов Отправки ON Отправки.КодОтправкиВагона = Участки.КодОтправкиВагона
446 
447 SELECT * FROM #UslData
448 ORDER BY Порядок
449 
450 IF object_id('tempdb..#UslData') IS NOT NULL DROP TABLE #UslData", id);

453  private static string sqlUslsGroup = @"
454 DECLARE @КодДокумента int
455 SET @КодДокумента = {0}
456 
457 SELECT X.*, Ед.ЕдиницаРус, НДС.СтавкаНдс,
458  CASE WHEN X.GUIDОказаннойУслуги IS NOT NULL AND EXISTS(SELECT 1 FROM НаборыУслуг (nolock) WHERE GuidОказаннойУслугиВходящей = X.GUIDОказаннойУслуги)
459  THEN 1 ELSE 0
460  END Перевыставлена,
461  CASE WHEN КодОказаннойУслуги IS NULL THEN КодРесурса ELSE КодОказаннойУслуги END КодГруппы,
462  CONVERT(varchar, @КодДокумента) + '_' + CONVERT(varchar, X.КодРесурса) + '_' + CONVERT(varchar,X.КодСтавкиНДС) + '_' + REPLACE(REPLACE(CONVERT(varchar,ЦенаБезНДС), '.','_'),',','_') УникальныйКлюч
463 FROM
464  (SELECT 1 Вагон, КодРесурса, РесурсРус, РесурсЛат, КодЕдиницыИзмерения, КодСтавкиНДС,
465  ЦенаБезНДС, SUM( ROUND( Всего/Количество, 2 ) ) ЦенаНДС, SUM( ROUND( Количество, 3 ) ) Количество,
466  SUM( СуммаБезНДС ) СуммаБезНДС, SUM( СуммаНДС ) СуммаНДС, SUM( Всего ) Всего,
467  0 Агент1, 0 Агент2, NULL GUIDОказаннойУслуги, NULL КодОказаннойУслуги
468  FROM vwОказанныеУслуги Услуги (nolock)
469  WHERE КодДокумента = @КодДокумента AND КодУчасткаОтправкиВагона IS NOT NULL
470  GROUP BY КодРесурса, РесурсРус, РесурсЛат, КодЕдиницыИзмерения, КодСтавкиНДС, ЦенаБезНДС
471  UNION ALL
472  SELECT 0 Вагон, КодРесурса, РесурсРус, РесурсЛат, КодЕдиницыИзмерения, КодСтавкиНДС,
473  ЦенаБезНДС, ROUND( Всего/Количество, 2 ) ЦенаНДС, ROUND( Количество, 3 ) Количество,
474  СуммаБезНДС, СуммаНДС, Всего,
475  Агент1, Агент2, GUIDОказаннойУслуги, КодОказаннойУслуги
476  FROM vwОказанныеУслуги Услуги (nolock)
477  WHERE КодДокумента = @КодДокумента AND КодУчасткаОтправкиВагона IS NULL) X
478  LEFT JOIN Справочники.dbo.ЕдиницыИзмерения Ед ON Ед.КодЕдиницыИзмерения = X.КодЕдиницыИзмерения
479  LEFT JOIN Справочники.dbo.СтавкиНДС НДС ON НДС.КодСтавкиНДС = X.КодСтавкиНдс
480 ORDER BY Вагон DESC";
481 
482  }
