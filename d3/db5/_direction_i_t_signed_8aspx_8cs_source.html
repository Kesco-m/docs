84  var cm = new SqlCommand(SQLQueries.SP_ВыполнениеУказанийIT);
85  cm.Connection = new SqlConnection(Config.DS_user);
86  cm.CommandType = CommandType.StoredProcedure;
87  cm.CommandTimeout = 0;
88  cm.Parameters.AddWithValue("@КодДокумента", int.Parse(Dir.Id));
89  cm.Parameters.Add("@RETURN_VALUE", SqlDbType.Int, 4);
90  cm.Parameters["@RETURN_VALUE"].Direction = ParameterDirection.ReturnValue;
91  try
92  {
93  cm.Connection.Open();
94  cm.ExecuteNonQuery();
95  if (cm.Parameters["@RETURN_VALUE"].Value != null
96  && !cm.Parameters["@RETURN_VALUE"].Value.Equals(DBNull.Value))
97  {
98  if (cm.Parameters["@RETURN_VALUE"].Value.ToString().Equals("1"))
99  RefreshData();
100  }
101  }
102  catch (Exception ex)
103  {
104  throw new DetailedException(ex.Message, ex, cm);
105  }
106  finally
107  {
108  if (cm != null && cm.Connection != null && cm.Connection.State.Equals(ConnectionState.Open))
109  cm.Connection.Close();
110  }
111  JS.Write("Wait.render(false);");
112  }
