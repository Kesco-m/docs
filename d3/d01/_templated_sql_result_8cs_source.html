90  public void SortResultColumn(string strColumn)
91  {
92  if (string.IsNullOrWhiteSpace(strColumn)) return;
93 
94  object sorObj = _stdAsc;
95 
96  for (int i = 0; i < sortData.Count; i++)
97  {
98  if (sortData[i].Key == strColumn)
99  {
100  if ((object)_stdAsc == sortData[i].Value)
101  sorObj = _stdDesc;
102 
103  sortData.RemoveAt(i);
104  break;
105  }
106  }
107 
108  sortData.Insert(0, new KeyValuePair<string, object>(strColumn, sorObj));
109 
110  Update();
111  }

118  public void SetSortMode(string strSortMode)
119  {
120  sortData.Clear();
121  string[] sa = strSortMode.Split(',');
122  foreach (string s in sa)
123  {
124  if (string.IsNullOrWhiteSpace(s)) continue;
125  string sort_column = s.Trim();
126 
127  object sorObj = _stdAsc;
128  int space_index = sort_column.LastIndexOf(' ');
129  if (space_index > -1)
130  {
131  if (0 == string.Compare(sort_column, space_index, _stdDesc, 0, _stdDesc.Length, true))
132  sorObj = _stdDesc;
133  else
134  if (0 != string.Compare(sort_column, space_index, _stdAsc, 0, _stdAsc.Length, true))
135  space_index = sort_column.Length;
136  }
137  else space_index = sort_column.Length;
138 
139  string strKey = sort_column.Substring(0, space_index).TrimEnd();
140  if (!sortData.Exists(p => { return p.Key == strKey; }))
141  sortData.Add(new KeyValuePair<string, object>(strKey, sorObj));
142  }
143  }
