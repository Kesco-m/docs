358  private static bool TryStrToAmount(this string str, out decimal amount)
359  {
360  amount = 0.0M;
361 
362  var val = str;
363  // Новая метода парсинга числа. Точка как и запятая воспринимается как
364  // разделитель дробной части, если этот же символ не используется как разделитель групп.
365  // Учитывается последний в строке символ, остальные вычищаются.
366  var decSep = CultureInfo.CurrentCulture.NumberFormat.NumberDecimalSeparator;
367  var groupSep = CultureInfo.CurrentCulture.NumberFormat.NumberGroupSeparator;
368 
369  if (groupSep != ".")
370  {
371  val = val.Replace(".", "*");
372  }
373 
374  if (groupSep != ",")
375  {
376  val = val.Replace(",", "*");
377  }
378 
379  var index = val.Trim().LastIndexOf("*", StringComparison.Ordinal);
380 
381  if (index != -1 && decSep.Length != 0)
382  {
383  var array = val.ToCharArray();
384  array[index] = decSep[0];
385  val = new string(array);
386  }
387 
388  val = val.Replace("*", "");
389  val = val.Replace(" ", "").Trim().ToLower();
390 
391  if (string.IsNullOrEmpty(val))
392  {
393  return false;
394  }
395 
396  var multiplier = 1.0M;
397 
398  if (val.EndsWith("k"))
399  {
400  multiplier = 1000.0M;
401  }
402  else if (val.EndsWith("m"))
403  {
404  multiplier = 1000000.0M;
405  }
406 
407  var chars = val.ToList();
408 
409  chars.RemoveAll(c => !(c == '-' || (decSep.Length != 0 && c == decSep[0]) || char.IsDigit(c)));
410 
411  val = new string(chars.ToArray());
412 
413 
414  val = val.Replace(".", ",");
415 
416  var format = CultureInfo.GetCultureInfo("ru").NumberFormat;
417 
418  amount = decimal.Parse(val, format);
419 
420  amount *= multiplier;
421 
422  return true;
423  }
