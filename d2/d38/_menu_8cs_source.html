148  private string ReturnActionHtml(MenuTree Item)
149  {
150  switch (Item.ActionType)
151  {
152  case MenuButtonActionType.UrlAction:
153  return
154  String.Format(
155  "onclick='window.open(\"{0}\", \"{1}\", \"menubar={2},location={3},resizable={4},scrollbars={5},status={6},height={7},width={8}\");' return false;",
156  Item.Action.Url,
157  Item.ItemID,
158  Item.Action.Menubar ? "1" : "0",
159  Item.Action.Location ? "1" : "0",
160  Item.Action.Resizable ? "1" : "0",
161  Item.Action.Scrollbars ? "1" : "0",
162  Item.Action.Status ? "1" : "0",
163  Item.Action.Height,
164  Item.Action.Width);
165  case MenuButtonActionType.CmdAction:
166  var cmdParametrs = "";
167  foreach (var param in Item.Action.CmdParams)
168  cmdParametrs = cmdParametrs + string.Format("\"{0}\", \"{1}\",", param.Key, param.Value);
169 
170  if (cmdParametrs != "") cmdParametrs = cmdParametrs.Remove(cmdParametrs.Length - 1, 1);
171  return String.Format("onclick='cmd(\"cmd\", \"{0}\"{1})'", Item.Action.CmdActionName,
172  !String.IsNullOrEmpty(cmdParametrs) ? "," + cmdParametrs : "");
173  case MenuButtonActionType.JSAction:
174  return String.Format("onclick='{0}'", Item.Action.Url);
175  case MenuButtonActionType.None:
176  return "";
177  }
178  return null;
