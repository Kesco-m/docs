401  public static List<DocType> GetDocTypesByNameAndTypes(string docTypeName, string typesStr = "")
402  {
403  string sqlWhere = string.Empty;
404  string sqlByNamePat = @"' ' + ТипДокумента LIKE N'%{0}%'";
405  string sqlByTypePat = @"КодТипаДокумента IN({0})";
406 
407  if (!string.IsNullOrEmpty(docTypeName) && !string.IsNullOrEmpty(typesStr))
408  {
409  sqlWhere = string.Concat(string.Format(sqlByNamePat, docTypeName), " AND ", string.Format(sqlByTypePat, typesStr));
410  }
411  if (!string.IsNullOrEmpty(docTypeName))
412  {
413  sqlWhere = string.Format(sqlByNamePat, docTypeName);
414  }
415  else if (!string.IsNullOrEmpty(typesStr))
416  {
417  sqlWhere = string.Format(sqlByTypePat, typesStr);
418  }
419 
420  string query;
421  //Все введенные символы могут оказаться недопустимыми и условие может остаться пустым
422  if(string.IsNullOrWhiteSpace(sqlWhere))
423  query = string.Format(SQLQueries.SELECT_ТипыДокументов, "");
424  else
425  query = string.Concat(string.Format(SQLQueries.SELECT_ТипыДокументов, ""), string.Format(" WHERE {0}", sqlWhere));
426 
427  return GetDocTypesList(query);
428  }
