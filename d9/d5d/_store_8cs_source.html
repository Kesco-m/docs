370  public int Save(bool isNew = false)

374  sqlParams.Add("@КодСклада", (isNew ? DBNull.Value : (object)Id));

381  sqlParams.Add("@КодХранителя", KeeperId == 0 ? DBNull.Value : (object)KeeperId);
382  sqlParams.Add("@КодРаспорядителя", ManagerCode == 0 ? DBNull.Value : (object)ManagerCode);
383  sqlParams.Add("@КодПодразделенияРаспорядителя", ((ManagerSubdivisionCode == 0) ? DBNull.Value : (object)ManagerSubdivisionCode));
384  sqlParams.Add("@КодДоговора", ((AgreementCode == 0) ? DBNull.Value : (object)AgreementCode));
385  sqlParams.Add("@Филиал", (StoreTypeId >= 10) ? string.Empty : Subsidiary);
386  sqlParams.Add("@Примечание", Note);
387 
388  var outputParams = new Dictionary<string, object>();
389  DBManager.ExecuteNonQuery(SQLQueries.SP_Лица_InsUpd_Склады, CommandType.StoredProcedure, Config.DS_person, sqlParams, outputParams);
390 
391  int storeID = 0;
392 
393  if (outputParams.ContainsKey("@RETURN_VALUE"))
394  storeID = (int)outputParams["@RETURN_VALUE"];
