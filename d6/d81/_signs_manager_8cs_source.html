279  public bool RemoveSign(string signId, bool ask)
280  {
281  var sign = _doc.DocSigns.First(i => i.Id == signId);
282 
283  if (sign.Unavailable)
284  {
285  // Подпись недоступна, возможно она уже удалена.
286  _docPage.ShowMessage(_resx.GetString("msgDelSignUnavaible"));
287  return false;
288  }
289 
290  if (ask && (sign.EmployeeId != _docPage.CurrentUser.EmployeeId || sign.SignType == 1))
291  {
292  var b = new StringBuilder();
293  var go = true;
294 
295  if (sign.SignType == 1) //подпись финальная
296  {
297  if (sign.EmployeeId != _docPage.CurrentUser.EmployeeId &&
298  sign.EmployeeInsteadOf != _docPage.CurrentUser.EmployeeId)
299  {
300  // сотрудник не совпадает
301  // Удаление завершающей подписи другого сотрудника невозможно.
302  b.Append(_resx.GetString("msgDelSignOtherEmplUnavaible"));
303  go = false;
304  }
305  else //сотрудник совпадает
306  {
307  var docTrans = Transaction.GetTransactionsByDocId(_doc.DocId);
308  if (docTrans.Count > 0) //есть транзакции
309  {
310  var t = docTrans[0];
311 
312  if ((t.CodeTypeGroup.Equals(1) && _docPage.User.IsInRole("74")) ||
313  (t.CodeTypeGroup.Equals(2) && _docPage.User.IsInRole("72")) ||
314  (t.CodeTypeGroup.Equals(3) && _docPage.User.IsInRole("76"))) //есть роль
315  {
316  b.Append(_resx.GetString("msgDelFinishSign"));
317  b.AppendFormat(_resx.GetString("msgDelFinishSign1"), docTrans.Count);
318  b.Append(_resx.GetString("msgDelFinishSign2"));
319  }
320  else //нет роли
321  {
322  b.Append(_resx.GetString("msgDelFinishSignWithTran"));
323  go = false;
324  }
325  }
326  else //нет транзакций
327  {
328  b.AppendFormat(_resx.GetString("msgDelFinishSign") + _resx.GetString("msgDelFinishSign2"));
329  }
330  }
331  }
332  else //подпись обычная
333  {
334  if (sign.EmployeeId != _docPage.CurrentUser.EmployeeId)
335  b.AppendFormat(_resx.GetString("msgDelSign"), sign.EmployeeFio);
336  else
337  {
338  b.AppendFormat(_resx.GetString("QMsgDelSign"));
339  }
340  }
341 
342  if (go)
343  {
344  _js.Write("ConfirmDeleteSign.render('{0}','{1}','{2}','{3}','{4}');",
345  _resx.GetString("msgOsnAttention0"),
346  HttpUtility.JavaScriptStringEncode(b.ToString()).Replace("\\r\\n", "<br>"),
347  _resx.GetString("QSBtnYes"), _resx.GetString("QSBtnNo"), sign.SignId);
348  //JS.Write("if(confirm('{0}')) cmd('cmd', 'RemoveSign','IdSign', '{1}','ask', '0');", b, signID);
349  }
350  else
351  {
352  _docPage.ShowMessage(b.ToString());
353  }
354  return false;
355  }
356 
357  var result = RemoveSign(sign);
358 
359  if (result)
360  {
361  var index = _doc.DocSigns.FindIndex(i => i.SignId == sign.SignId);
362  if (index != -1)
363  _doc.DocSigns.RemoveAt(index);
364 
365  var last = _doc.DocSigns.LastOrDefault();
366  if (last != null)
367  last.CanDelete = 1;
368  }
369  else
370  return false;
371 
372 
373  return true;
374  }
