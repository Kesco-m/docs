45  public string RenderSqlWhereClause()
46  {
47  /* в зависимости от количества слов в поиске определяем поля для поиска.
48  * 1 слово - фамилия | имя
49  * 2 слова - фамилия + имя | имя + фамилия | имя + отчество
50  * 3 слова - фамилия + имя + отчество | имя + отчество + фамилия
51  * 4+ слов - не накладываем ограничений
52  * к указанным полям добавляются соответствующие английские поля ("Иван" + " " + "Ivan") - в полученной строке ищем */
53  string result = "";
54  switch (_wordsCount)
55  {
56  case 1:
57  result = "(";
58  if (_checkID > 0)
59  {
60  result += String.Format("(T0.КодСотрудника = {0} OR T0.КодСотрудника IN (SELECT КодСотрудника FROM @TblTel X)) OR ", _checkID);
61  }
62  result += "T0.ФамилияRL LIKE @S1 OR T0.LastName LIKE @S1 OR T0.ИмяRL LIKE @S1 OR T0.FirstName LIKE @S1";
63  result += ")";
64  break;
65  case 2:
66  result += @"(
67  ( (T0.ФамилияRL LIKE @S1 OR T0.LastName LIKE @S1) AND (T0.ИмяRL LIKE @S2 OR T0.FirstName LIKE @S2) ) OR
68  ( (T0.ИмяRL LIKE @S1 OR T0.FirstName LIKE @S1) AND (T0.ФамилияRL LIKE @S2 OR T0.LastName LIKE @S2) ) OR

74  ( (T0.ИмяRL LIKE @S1 OR T0.FirstName LIKE @S1) AND (T0.ОтчествоRL LIKE @S2 OR T0.MiddleName LIKE @S2) AND (T0.ФамилияRL LIKE @S3 OR T0.LastName LIKE @S3) ) )";
