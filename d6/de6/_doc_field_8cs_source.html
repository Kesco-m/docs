258  get
259  {
260  var val = Value;
261  if (val == null)
262  return true;
263 
264  var type = val.GetType();
265 
266  switch (type.Name)
267  {
268  case "String":
269  if (string.IsNullOrEmpty((string) val))
270  return true;
271  break;
272  case "Int32":
273  var i = (int?) val;
274  if (i == 0)
275  return true;
276  break;
277  case "Decimal":
278  var m = (decimal?)val;
279  if (m == 0)
280  return true;
281  break;
282  case "DateTime":
283  var dt = (DateTime?) val;
284  if (dt == DateTime.MinValue)
285  return true;
286  break;
287  case "Byte":
288  var b = (byte?) val;
289  if (b == 0)
290  return true;
291  break;
292  case "Double":
293  var d = (double?) val;
294  if (d == 0)
295  return true;
296  break;
297  default:
298  throw new ArgumentException("Не реализовано для типа: " + type.Name);
299  }
300 
301  return false;
302  }
