623  static string sqlGetMrisSales = @"
624 SELECT X.*, Ед.ЕдиницаРус, НДС.СтавкаНдс,
625 CONVERT(varchar, {0}) + '_' + CONVERT(varchar, X.КодРесурса) + '_' + CONVERT(varchar, X.КодСтавкиНДС) + '_' + REPLACE(REPLACE(CONVERT(varchar,ЦенаБезНДС), '.','_'),',','_') УникальныйКлюч
626 FROM(
627  SELECT
628  КодРесурса
629  ,РесурсРус
630  ,РесурсЛат
631  ,SUM(ROUND(Количество,3)) Количество
632  ,КодЕдиницыИзмерения
633  ,ЦенаБезНдс
634  ,КодСтавкиНдс
635  ,SUM(ROUND(Всего/ISNULL(NULLIF(Количество,0),1),2)) ЦенаНдс
636  ,SUM(СуммаБезНдс) СуммаБезНдс
637  ,SUM(СуммаНдс) СуммаНдс
638  ,SUM(Всего) Всего
639  FROM vwДвиженияНаСкладах (nolock)
640  WHERE КодДокумента = {0}
641  GROUP BY КодРесурса, РесурсРус, РесурсЛат, КодЕдиницыИзмерения, КодСтавкиНдс, ЦенаБезНдс
642 ) X
643  LEFT JOIN Справочники.dbo.ЕдиницыИзмерения Ед ON Ед.КодЕдиницыИзмерения = X.КодЕдиницыИзмерения
644  LEFT JOIN Справочники.dbo.СтавкиНДС НДС ON НДС.КодСтавкиНДС = X.КодСтавкиНдс";
645  }
646 }
