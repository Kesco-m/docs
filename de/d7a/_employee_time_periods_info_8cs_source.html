600  private void CompletePeriods()
601  {
602  TimeSpan ts = _endTime.AddDays(1) - _startTime;
603  double totalDays = ts.TotalDays; // число дней в диапазоне дат
604  TimeSpan t = _startTime - _startTime;
605  _hasErrors = false;
606 
607  //if (_sourceRows.Length > 0)
608  if (_sourceEvents.Length < 1) return;
609 
610  _workPeriods.Clear();
611  _absentPeriods.Clear();
612 
613  for (double dayNum = 0; dayNum < totalDays; dayNum++)
614  { // нарезка диапазона по суткам
615  bool expectingEntrance = true;
616  DateTime enterTime = DateTime.MinValue;
617  bool exitTimeEmpty = true;
618  DateTime exitTime = DateTime.MinValue;
619  int enterCode = -1;
620  string enterPlace = String.Empty;
621  bool isExitNextDay = false;
622  bool isEnterTomorow = false;
623  int firstEnterTomorow = -1;
624 
625  DateTime nextDay = _startTime.AddDays(dayNum + 1);
626  DateTime currentDay = _startTime.AddDays(dayNum);
627 
628  for (int i = 0; i < _sourceEvents.Length; i++)
629  {
630  if (_sourceEvents[i].when > nextDay.AddDays(1)) break;
631 
632  DateTime rowExitTime = _sourceEvents[i].when;
633 
634  bool fEnterCodeIsNull = !_sourceEvents[i].code.HasValue;
635  bool fGreaterThanNext = rowExitTime >= nextDay;
636  bool fLessThanCurrent = rowExitTime < currentDay;
637 
638 
639  if (fLessThanCurrent || fGreaterThanNext) /* если запись не попадает в диапазон обрабатываемого дня */
640  {
641  if (fGreaterThanNext && fEnterCodeIsNull)
642  {
643  isExitNextDay = true;
644  if (!isEnterTomorow)
645  {
646  firstEnterTomorow = i;
647  }
648  }
649  else if (fGreaterThanNext && !fEnterCodeIsNull)
650  {
651  if (firstEnterTomorow == -1)
652  {
653  isEnterTomorow = true;
654  firstEnterTomorow = i;
655  }
656  }
657  else if (fLessThanCurrent && !fEnterCodeIsNull)
658  {
659  IsEnterPrevDayByDay = true;
660  enterCode = (int)_sourceEvents[i].code;
661  enterPlace = _sourceEvents[i].place;
662  }
663  else if (fLessThanCurrent && fEnterCodeIsNull)
664  {
665  IsEnterPrevDayByDay = false;
666  }
667  continue;
668  }
669 
670  if (expectingEntrance)
671  {
672  if (fEnterCodeIsNull)
673  {
674  if (IsEnterPrevDayByDay)
675  {
676  exitTime = rowExitTime;
677  _workPeriods.Add(new TimeInterval(currentDay.AddMinutes((int)t.TotalMinutes), exitTime, enterCode, false, false, enterPlace));
678  exitTimeEmpty = false;
679  IsEnterPrevDayByDay = false;
680  isExitNextDay = false;
681  continue;
682  }
683  _hasErrors = true;
684  // исключительная ситуация: если не был зафиксирован вход (идут 2 выхода подряд или 1ая позиция - выход)
685  if (exitTimeEmpty)
686  {
687  // 1ая позиция - выход
688  if (_employeePrimary)
689  {
690  _workPeriods.Add(new TimeInterval(currentDay.AddMinutes((int)t.TotalMinutes), rowExitTime, -1, true, false));
691  }
692  else
693  {
694  _workPeriods.Add(new TimeInterval(rowExitTime, rowExitTime, -1, true, false));
695  }
696  }
697  else
698  {
699  // идут 2 выхода подряд
700  if (_employeePrimary)
701  {
702  _workPeriods.Add(new TimeInterval(exitTime, rowExitTime, -1, true, false));
703  _absentPeriods.Add(new TimeInterval(exitTime, exitTime, -1, false, true));
704  }
705  else
706  {
707  _workPeriods.Add(new TimeInterval(rowExitTime, rowExitTime, -1, true, false));
708  _absentPeriods.Add(new TimeInterval(exitTime, rowExitTime, -1, false, true));
709  }
710  }
711 
712  exitTimeEmpty = false;
713  exitTime = rowExitTime;
714  continue;
715  }
716 
717  enterCode = (int)_sourceEvents[i].code;
718  enterTime = rowExitTime;
719  enterPlace = _sourceEvents[i].place;
720  expectingEntrance = false;
721 
722  if (!exitTimeEmpty)
723  {
724  _absentPeriods.Add(new TimeInterval(exitTime, enterTime, -1));
725  }
726  }
727  else
728  {
729  if (!fEnterCodeIsNull)
730  {
731  // исключительная ситуация: если не был зафиксирован выход (идут 2 входа подряд)
732  _hasErrors = true;
733 
734  if (_employeePrimary)
735  {
736  _workPeriods.Add(new TimeInterval(enterTime, rowExitTime, enterCode, false, true, enterPlace));
737  _absentPeriods.Add(new TimeInterval(rowExitTime, rowExitTime, -1, true, false));
738  }
