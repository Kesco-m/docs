537  private void SaveBaseDocs(List<DBCommand> cmds = null)
538  {
539  if(_baseDocs == null) return;
540 
541  var origBaseDocs = DocLink.LoadBasisDocsByChildId(DocId);
542 
543  //докуметы после удаления(оставшиеся)
544  var origDelDocs = new List<DocLink>(origBaseDocs.Count);
545 
546  // сохраняем документы в случае отката
547  var delLinks = new Stack<DocLink>(origBaseDocs.Count);
548  var newLinks = new Stack<DocLink>(origBaseDocs.Count);
549 
550  try
551  {
552  // удаление
553  foreach (var o in origBaseDocs)
554  {
555  // если в новой "версии" такого нет, то удаляем из базы
556  if (!_baseDocs.Exists(b => b.Id == o.Id))
557  {
558  o.Delete(cmds);
559  delLinks.Push(o.Clone());
560  continue;
561  }
562 
563  origDelDocs.Add(o);
564  }
565  }
566  catch (Exception e)
567  {
568  // возвращаем в первоначальнее состояние
569  while (delLinks.Count > 0)
570  {
571  var d = delLinks.Pop();
572  d.Create(cmds);
573  }
574 
575  DetailedException dex = new DetailedException(e.Message, e);
576  Logger.WriteEx(dex);
577  throw dex;
578  }
579 
580  try
581  {
582  // добавляем
583  foreach (var i in _baseDocs)
584  {
585  // Если в базе такого нет, то добавляем
586  if (!origDelDocs.Exists(o => o.DocFieldId == i.DocFieldId && o.BaseDocId == i.BaseDocId))
587  {
588  i.SequelDocId = DocId;
589  i.Create(cmds);
590  newLinks.Push(i.Clone());
591  }
592  }
593  }
594  catch (Exception e)
595  {
596  // возвращаем в первоначальнее состояние
597  while (delLinks.Count > 0)
598  {
599  var d = delLinks.Pop();
600  d.Create(cmds);
601  }
602  while (newLinks.Count > 0)
603  {
604  var i = newLinks.Pop();
605  i.Delete(cmds);
606  }
