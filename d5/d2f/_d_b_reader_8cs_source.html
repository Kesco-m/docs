150  private void ExecuteReader(string comdText, CommandType type, Dictionary<string, object> parameters, Dictionary<string, object> parametersOut)
151  {
152  try
153  {
154  using (_command = new SqlCommand(comdText, _connect) { CommandType = type })
155  {
156 
157  if (parameters != null)
158  foreach (var p in parameters)
159  _command.Parameters.AddWithValue(p.Key, p.Value ?? DBNull.Value);
160 
161  if (parametersOut != null)
162  foreach (var p in parametersOut)
163  {
164  var par = _command.Parameters.AddWithValue(p.Key, p.Value);
165 
166  par.Direction = ParameterDirection.Output;
167  par.Size = -1;
168  }
169 
170  if (type == CommandType.StoredProcedure)
171  {
172  if (parametersOut == null)
173  parametersOut = new Dictionary<string, object> { { "@RETURN_VALUE", -1 } };
174  else
175  parametersOut.Add("@RETURN_VALUE", -1);
176 
177  _command.Parameters.Add(new SqlParameter
178  {
179  Direction = ParameterDirection.ReturnValue,
180  ParameterName = "@RETURN_VALUE"
181  });
182  }
183 
184  _connect.Open();
185  _reader = _command.ExecuteReader(CommandBehavior.CloseConnection);
186  }
187  }
188  catch (Exception ex)
189  {
190  DetailedException dex = new DetailedException(ex.Message, ex, _command);
191  Logger.WriteEx(dex);
192  Dispose();
193  throw dex;
194  }
