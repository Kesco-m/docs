196  public static void CheckUniqueEmail(EntityPage page, TextWriter w, Direction dir)

206  var result = ADSI.FindAccountByEmail(email);
207  if (result == null)
208  {
209  w.Write("");
210  return;
211  }
212  var empl = dir.Sotrudnik;
213  var emailUsersGuid = new StringCollection();
214  for (var i = 0; i < result.Properties["objectGUID"].Count; i++)
215  {
216  if (empl.Guid.ToString() != result.Properties["objectGUID"][i].ToString())
217  emailUsersGuid.Add(result.Properties["objectGUID"][i].ToString());
218  }
219  if (emailUsersGuid.Count == 0)
220  {
221  w.Write("");
222  return;
223  }
224 
225  var emailAddress = new StringCollection();
226  for (var i = 0; i < result.Properties["proxyAddresses"].Count; i++)
227  {
228  if (result.Properties["proxyAddresses"][i].ToString()
229  .StartsWith("SMTP", StringComparison.CurrentCultureIgnoreCase))
230  emailAddress.Add(result.Properties["proxyAddresses"][i].ToString().Substring(5).Trim());
231  }
232 
233  if (!emailAddress.Contains(email))
234  {
235  w.Write("");
236  return;
237  }
238 
239  w.Write(LocalResx.GetString("_NTF_EmailNotUnique"));
