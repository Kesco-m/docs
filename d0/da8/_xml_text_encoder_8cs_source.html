55  {
56  const int endSentinel = -1;
57  while (_buf.Count == 0 && _source.Peek() != endSentinel)
58  {
59  // Strings in .NET are assumed to be UTF-16 encoded [1].
60  var c = (char)_source.Read();
61 // if (Entities.ContainsKey(c))
62 // {
63 // // Encode all entities defined in the XML spec [2].
64 // foreach (var i in Entities[c]) _buf.Enqueue(i);
65 // }
66 // else !!!
67  if (!(0x0 <= c && c <= 0x8) &&
68  c != (char)0xB && c != (char)0xC && c != 0xFFFE && c != 0xFFFF &&
69  !(0xE <= c && c <= 0x1F) &&
70  !(0x7F <= c && c <= 0x84) &&
71  !(0x86 <= c && c <= 0x9F) &&
72  !(0xD800 <= c && c <= 0xDFFF) )
73  {
74  // Allow if the Unicode codepoint is legal in XML [3].
75  _buf.Enqueue(c);
76  }
77  else if (char.IsHighSurrogate(c) &&
78  _source.Peek() != endSentinel &&
79  char.IsLowSurrogate((char)_source.Peek()))
80  {
81  // Allow well-formed surrogate pairs [1].
82  _buf.Enqueue(c);
83  _buf.Enqueue((char)_source.Read());
84  }
85  else if (!_filterIllegalChars)
86  {
87  // Note that we cannot encode illegal characters as entity
88  // references due to the "Legal Character" constraint of
89  // XML [4]. Nor are they allowed in CDATA sections [5].
90  throw new ArgumentException(
91  String.Format("Illegal character: '{0:X}'", (int)c));
92  }
93  }
