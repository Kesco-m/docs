258  public static string CreatePredicate(int pred_type, string column, string value)
259  {
260  if (0 == pred_type) return null;
261  if (pred_type > 10000) return null; //эти значения используются для пользовательских типов фильтрации
262 
263  Func<int, string, string> ResultFunc = (t, r) =>
264  {
265  if (t % 2 == 0 && r.Length > 0) return "NOT (" + r + ")";
266  return r;
267  };
268 
269  //условия фильтрациии для кодов перечисленных через запятую, value - это имя параметра
270  if (pred_type > 0 && pred_type < 5)
271  {
272 
273  //Элементы из списка
274  //Элементы за исключением
275  if (pred_type < 3)
276  {
277  if (null != value && value.Trim().Length > 0)
278  return ResultFunc(pred_type, string.Format("{0} IN(SELECT value FROM Инвентаризация.dbo.fn_SplitInts({1}))", column, value));
279  }
280 
281  //Указано какое-либо значение
282  //Значение не указано
283  if (pred_type < 5)
284  return ResultFunc(pred_type, string.Format("{0} IS NOT NULL", column));
285  }
286 
287  //условия фильтрациии для текстовых полей, value - это строка из нескольких слов
288  if (pred_type > 10 && pred_type < 25)
289  {
290  //Слова разделяются следующими символами: пробел, неразрывный пробел, табуляция, перевод строки, возврат каретки,
291  //одинарная кавычка, двойная кавычка или символы из строки "_[]{}()/\%@&$:;!?".
292  string clear_value = Regex.Replace(value, "[\xa0\t\n\r'`\\\"_\\[\\]{}()/\\\\%@&$:;!?]", " ");
293  clear_value = Regex.Replace(clear_value, "\\s+", " ");
294  clear_value = clear_value.Trim();
295 
296  if (pred_type < 23 && clear_value.Length < 1) return null; //не допускаем появление условий типа LIKE '%%' или LIKE '%'
297 
298  //Содержит все слова
299  //Не содержит какого-либо слова
300  if (pred_type < 13) return ResultFunc(pred_type, column + " LIKE '%" + clear_value.Replace(" ", "%' AND " + column + " LIKE '%") + "%'");
301 
302  //Содержит все слова в заданном порялке
303  //Не содержит все слова в заданном порялке
304  if (pred_type < 15) return ResultFunc(pred_type, column + " LIKE '%" + clear_value.Replace(' ', '%') + "%'");
305 
306  //Содержит какое-либо слово
307  //Не содержит ни одного слова
308  if (pred_type < 17) return ResultFunc(pred_type, column + " LIKE '%" + clear_value.Replace(" ", "%' OR " + column + " LIKE '%") + "%'");
309 
310  //Слова начинаются
311  if (pred_type < 18) return ResultFunc(pred_type, "(' ' + " + column + ") LIKE '% " + clear_value.Replace(" ", "%' AND (' ' + " + column + ") LIKE '% ") + "%'");
312 
313  //Слова не начинаются (к условию будет применен NOT)
314  if (pred_type < 19) return ResultFunc(pred_type, "(' ' + " + column + ") LIKE '% " + clear_value.Replace(" ", "%' OR (' ' + " + column + ") LIKE '% ") + "%'");
315 
316  //Слова заканчиваются
317  if (pred_type < 20) return ResultFunc(pred_type, "(" + column + " + ' ') LIKE '%" + clear_value.Replace(" ", " %' AND (" + column + " + ' ') LIKE '%") + " %'");
318 
319  //Слова не заканчиваются (к условию будет применен NOT)
320  if (pred_type < 21) return ResultFunc(pred_type, "(" + column + " + ' ') LIKE '%" + clear_value.Replace(" ", " %' OR (" + column + " + ' ') LIKE '%") + " %'");
321 
322  //Полностью совпадает
323  //Полностью не совпадает
324  if (pred_type < 23) return ResultFunc(pred_type, column + "= '" + clear_value + "'");
325 
326  //Пустая строка
327  //Не пустая строка
328  if (pred_type < 25) return ResultFunc(pred_type, column + "=''");
329  }
330 
331  //условия фильтрациии для дат, value - это имя параметра
332  if (pred_type > 50 && pred_type < 55)
333  {
334  //После указанной даты
335  //До указанной даты
336  if (pred_type < 53) return ResultFunc(pred_type, column + ">" + value);
337 
338  //Указана какая-либо дата
339  //Дата не указана
340  if (pred_type < 55) return ResultFunc(pred_type, column + " IS NOT NULL");
341  }
342 
343  return null;
344  }
