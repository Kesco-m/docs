155  private string RenderBodyConrol(out bool displayControl, bool commetMessage = false)
156  {
157  var sb = new StringBuilder();
158  var ret = "";
159  displayControl = false;
160  List<CometClient> cometClients = new List<CometClient>();
161  foreach (
162  var page in
163  _entityActiveConnections.OrderByDescending(x => x.Start)
164  .ToList()
165  .Select(state => V4Page.Application[state.ClientGuid])
166  .OfType<Page>())
167  {
168  using (var wr = new StringWriter())
169  {
170  if (page.CurrentUser.Id == V4Page.CurrentUser.Id) continue;
171 
172  var client = cometClients.FirstOrDefault(x => x.IdEmployee == page.CurrentUser.Id);
173 
174  if (client != null)
175  {
176  if (!client.IsEditable && page.IsEditable) client.IsEditable = page.IsEditable;
177  }
178  else
179  {
180  V4Page.RenderLinkEmployee(wr, Guid.NewGuid().ToString(), page.CurrentUser, NtfStatus.Empty);
181  cometClients.Add(new CometClient { IdPage = page.IDPage, IdEmployee = page.CurrentUser.Id, EmployeeLink = wr.ToString(), IsEditable = page.IsEditable });
182  }
183  }
184  if (!displayControl) displayControl = true;
185  }
186 
187  foreach (var x in cometClients)
188  {
189  sb.AppendFormat("<div>{0} - {1}</div>",
190  x.EmployeeLink,
191  x.IsEditable ? _lblEditable : _lblBrowse);
192  }
193 
194  if (commetMessage)
195  {
196  var cometScript = "";
197  ret = sb.ToString();
198  if (string.IsNullOrEmpty(ret))
199  {
200  cometScript = string.Format("$('#{0}_0').hide(); v4_cometShowImgMsg(0); v4_cometCloseUsersList();",
201  HtmlID);
202  }
203  else
204  {
205  cometScript = string.Format("$('#{0}_0').show(); v4_setToolTip();", HtmlID);
206  displayControl = true;
207  }
208  ret = string.Format("<js>$('#{0}_1_Body').html('{1}');{2}</js>", HtmlID,
209  HttpUtility.JavaScriptStringEncode(ret),
210  cometScript);
211  }
212  else
213  ret = sb.ToString();
214 
215 
216  return ret;
217  }
