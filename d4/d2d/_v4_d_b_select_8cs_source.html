144  private string GetCallbackUrl()
145  {
146  string callbackUrl = "";
147  if (V4Page.V4Request.UrlReferrer != null)
148  {
149  string[] path = V4Page.V4Request.UrlReferrer.OriginalString.Split(new string[] { "?" }, StringSplitOptions.RemoveEmptyEntries);
150  string query = "";
151  if (path.Length > 1)
152  {
153  string keys = "";
154  string[] qs = path[1].Split(new[] { "&" }, StringSplitOptions.RemoveEmptyEntries);
155  foreach (string s in qs)
156  {
157  string[] arrKeys = s.Split(new[] { "=" }, StringSplitOptions.RemoveEmptyEntries);
158  if (arrKeys.Length > 0)
159  keys += arrKeys[0] + "&";
160  }
161  qs = keys.Split(new[] { "&" }, StringSplitOptions.RemoveEmptyEntries);
162  query = qs.Where(s => s != "type" && s != "currentperson" && s != "docDir").Aggregate("?", (current, s) => current + (s + "=" + V4Page.V4Request.QueryString[s] + "&"));
163  }
164  else
165  {
166  query = "?";
167  }
168  callbackUrl = path[0] + query + "idp=" + V4Page.IDPage + "&ctrl=" + ID + "&cmd=callback";
169  }
170  return callbackUrl;
171  }
