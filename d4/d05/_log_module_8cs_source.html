139  {

143  catch { }
144  }
145 
146  if( ex == null ) return;
147 
148  XmlElement xml = GetErrorXml( ex );
149 
150  if( ((XmlElement)xml.SelectSingleNode( "Info" )).GetAttribute( "mail" ) == "1" )
151  {
152  // Заголовок - используется для автоматической разборки писем на сервере
153  string subject = "";
154 
155  string pr = ((XmlElement)xml.SelectSingleNode( "Info/Node[@id='Priority']" )).GetAttribute( "value" );
156  if (pr == Enum.GetName(typeof(Priority), Priority.Error))
157  {
158  subject = "AppError";
159 
160  // Доп. проверка на ExternalError
161  pr = GetPriority(ex, Priority.Error).ToString();
162  ((XmlElement)xml.SelectSingleNode("Info/Node[@id='Priority']")).SetAttribute("value", pr);
163  }
164  if (pr == Enum.GetName(typeof(Priority), Priority.ExternalError))
165  subject = "Error";
166  else if (pr == Enum.GetName(typeof(Priority), Priority.Info) || pr == Enum.GetName(typeof(Priority), Priority.Alarm))
167  subject = "*" + pr + "* ";
168 
169  XmlElement component = (XmlElement)xml.SelectSingleNode( "Info/Node[@id='Component']" );
170  string build = component != null ? component.GetAttribute( "value" ) : "";
171 
172  subject += String.Format( "[{0}]{1}:{2}", _appName,
173  ( build != "" ? "[" + build + "]" : "" ),
174  ((XmlElement)xml.SelectSingleNode("Info")).GetAttribute("message")).Replace("\n", " ").Replace("\r", " ");
175 
176  if( subject.Length > MaxSubjLen )
177  subject = subject.Substring( 0, MaxSubjLen );
178 
179  XmlElement func = ((XmlElement)xml.SelectSingleNode("Info/Node[@id='Function']"));
180  XmlElement host = ((XmlElement)xml.SelectSingleNode("Info/Node[@id='Computer']"));
181  XmlElement user = ((XmlElement)xml.SelectSingleNode("Info/Node[@id='Computer']"));

183  lock (_sentMessages.SyncRoot)
184  {
185  try
186  {

191  catch { }
192  }

211  }

232  }

277 
278  // Сообщение и тип оригинального исключения
279  if( exOrder == 0 )
280  {

282  message = ex.Message;

288  priority = Enum.GetName(typeof (Priority), Priority.Error);

301  else if (ex is LogicalException)

304  if( (int)(ex as LogicalException).PriorityLevel < 1 )
305  priority = Enum.GetName(typeof (Priority), Priority.Error);
306  }
307  else

309  }

312  if( ex is SqlException )

314  else if( ex is DetailedException )
315  sendMail = ( ex as DetailedException ).SendMail;

317  sendMail = true;
318  }
319  }

326

329  else

331  }

342  }

353  if( declaringType != null )

355  string componentName = declaringType.Assembly.GetName( false ).Name;
356  if( kescoBuilds.IndexOf( componentName ) != -1 )
357  {
358  build = declaringType.Assembly.GetName( false ).Name;
359  version = declaringType.Assembly.GetName().Version.ToString();
360  method = m.Name;

362  }
363  }
364  }
365  }

861  private static Priority GetPriority(Exception ex, Priority pr)
862  {
863  if (ex == null)
864  return pr;
865 
866  while (ex.InnerException != null)
867  ex = ex.InnerException;
868 
869  if (ex is SqlException && Enum.IsDefined(typeof(SqlErrors), ((SqlException)ex).Number))
870  return Priority.ExternalError;
871 
872  if (ex.Message.Contains("provider: Named Pipes Provider, error: 40") || ex.Message.Contains("provider: Поставщик именованных каналов, error: 40"))
873  return Priority.ExternalError;
874 
875  if (ex is SqlException && ex.StackTrace.Contains("System.Data.SqlClient.SqlConnection.Open()"))
876  return Priority.ExternalError;
877 
878  if (ex is System.Net.WebException && ((System.Net.WebException)ex).Status == WebExceptionStatus.ProtocolError)
879  {
880  var response = ((System.Net.WebException)ex).Response as HttpWebResponse;
881  if (response != null && (int)response.StatusCode >= 500)
882  return Priority.ExternalError;
883  }
884 
885  int hr = Marshal.GetHRForException(ex);
886  if (hr == HRESULTs.COR_E_NOTSUPPORTED || hr == HRESULTs.COR_E_UNAUTHORIZEDACCESS || hr == HRESULTs.COR_E_TIMEOUT)
887  return Priority.ExternalError;
888 
889  if (ex is SqlException && (/*hr == HRESULTs.COR_E_LOGIN_FAILED ||*/ hr == HRESULTs.COR_E_LOGIN_FAILED1))
890  return Priority.ExternalError;
891 
892  if (ex is System.Runtime.InteropServices.COMException && hr == HRESULTs.COR_E_SRV_NOT_OPERATIONAL)
893  return Priority.ExternalError;
894 
895  if (ex is System.Web.Services.Protocols.SoapException && ex.StackTrace.Contains("ReportingService") && hr == HRESULTs.COR_E_SRV_REFUSED)
896  return Priority.ExternalError;
897 
898  if (ex is System.Net.Sockets.SocketException && Enum.IsDefined(typeof(TcpErrors), ((System.Net.Sockets.SocketException)ex).ErrorCode))
899  return Priority.ExternalError;
900 
901  if (ex is System.IO.IOException && hr == HRESULTs.COR_E_NET_NOT_REACHED)
902  return Priority.ExternalError;
903 
904  if (ex is System.Net.WebException)
905  {
906  System.Net.WebException we = ex as System.Net.WebException;
907  if (we.Status != System.Net.WebExceptionStatus.ProtocolError && we.Status != System.Net.WebExceptionStatus.UnknownError)
908  return Priority.ExternalError;
909  }
910 
911  return pr;
912  }
