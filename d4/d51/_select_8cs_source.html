712  public override void ProcessCommand(NameValueCollection collection)
713  {
714  base.ProcessCommand(collection);
715 
716  if (collection["checked"] != null)

718  bool is_checked = collection["checked"] == "1";
719  if (_checked != is_checked)
720  {
721  Checked = is_checked;
722  }
723 
724  return;
725  }
726 
727  if (collection["vn"] != null)
728  {
729  var tn = "";
730  if (collection["tn"] != null) tn = collection["tn"].Trim();
731  SetControlValue(collection["vn"], tn);
732  ChangeProperty("Value");
733  }
734  else if (collection["cmd"] != null)
735  {
736  switch (collection["cmd"])
737  {
738  case "popup":
739  string st_param = collection["st"];
740  SearchText = st_param == null ? string.Empty : st_param.Trim();
741 
742  string tn_param = collection["tn"];
743  string clientValueText = tn_param==null ? string.Empty : tn_param.Trim();
744 
745  if (clientValueText != SearchText && SearchText.Length > 0)
746  {
747  Value = "";
748  ValueText = SearchText;
749  }
750  ShowPopupWindow();
751  break;
752 
753  case "btn":
754  if (Value.Length == 0 || SearchText.Length > 0 || URLShowEntity.Length == 0)
755  {
756  SearchText = collection["tn"].Trim();
757  Value = "";
758  ValueText = SearchText;
759  ShowPopupWindow();
760  }
761  break;
762  case "Refresh":
763  SetControlValue(Value, ValueText);
764  break;
765  case "popupHead":
766  ShowPopupWindowClause();
767  break;
768  case "setHead":
769  SetControlClause(collection["val"]);
770  break;
771  case "RemoveSelectedItem":
772  if (ConfirmRemove && collection["ask"] == null || collection["ask"] == "1")
773  RemoveItemComfirm(collection["id"]);
774  else
775  RemoveSelectedItem(collection["id"]);
776  break;
777  case "clearSelectedItems":
778  ClearSelectedItems();
779  break;
780  }
781  }
782  else if (collection["tn"] != null)
783  {
784  if (string.IsNullOrEmpty(collection["tn"]))
785  {
786  var oldVal = Value;
787  Value = "";
788  ValueText = "";
789  OnChanged(new ProperyChangedEventArgs(oldVal, collection["vn"]));
790  }
791  else
792  {
793  SearchText = collection["tn"].Trim();
794  Value = "";
795  ValueText = SearchText;
796  ShowPopupWindow();
797  if (ValueText != collection["tn"])
798  SetPropertyChanged("ValueText");
799  }
