161  public DataTable GetPostSupervisor()
162  {
163  DataTable dt = new DataTable();
164  if (EmployeeFrom.ValueString.Length == 0) return dt;
165  string sql = @"
166 DECLARE @Сотрудник nvarchar(300), @ДолжностьСотрудника nvarchar(300),
167  @LСотрудника int, @RСотрудника int,
168  @КодЛицаРуководителя int, @КодСотрудникаРуководителя int, @Руководитель nvarchar(300), @ДолжностьРуководителя nvarchar(300)
169 
170 SELECT @Сотрудник=Я.Сотрудник,
171  @ДолжностьСотрудника=МояДолжн.Должность,
172  @LСотрудника=МояДолжн.L,
173  @RСотрудника=МояДолжн.R,
174  @КодСотрудникаРуководителя=Рук.КодСотрудника,
175  @КодЛицаРуководителя=Рук.КодЛица,
176  @Руководитель=Рук.ФИО,
177  @ДолжностьРуководителя=РукДолжн.Должность
178 FROM Сотрудники Я
179  INNER JOIN vwДолжности МояДолжн ON Я.КодСотрудника = МояДолжн.КодСотрудника
180  INNER JOIN vwДолжности РукДолжн ON МояДолжн.Parent = РукДолжн.КодДолжности
181  LEFT OUTER JOIN Сотрудники Рук ON РукДолжн.КодСотрудника = Рук.КодСотрудника
182 WHERE (Я.КодСотрудника = " + EmployeeFrom.ValueString + @") AND (МояДолжн.Совместитель = 0)
183 
184 IF @@ROWCOUNT>0 AND @КодСотрудникаРуководителя IS NULL
185  SELECT TOP 1 @КодЛицаРуководителя=S.КодЛица, @КодСотрудникаРуководителя=S.КодСотрудника,
186  @Руководитель=S.ФИО, @ДолжностьРуководителя=Должность
187  FROM vwДолжности D
188  INNER JOIN Сотрудники S ON D.КодСотрудника=S.КодСотрудника
189  WHERE L<@LСотрудника AND R>@RСотрудника AND D.КодСотрудника IS NOT NULL
190  ORDER BY L DESC
191 
192 SELECT @Сотрудник Я, @ДолжностьСотрудника МояДолжность, @КодСотрудникаРуководителя КодСотрудникаРуководителя,
193  @КодЛицаРуководителя КодЛица, @Руководитель Руководитель, @ДолжностьРуководителя ДолжностьРуководителя
194 ";
195  SqlDataAdapter da = new SqlDataAdapter(sql,Kesco.Lib.Web.Settings.Config.DS_user);
196  da.Fill(dt);
197 
198  return dt;
199  }
