109  }

112  sb.Remove(sb.Length - 1, 1);

121  private string GetDocTypeIDsStr()
122  {
123  if (DocTypeParams.Count > 0)
124  {
125  var typeIDs = new StringCollection();
126 
127  var selectLRSql = "SELECT КодТипаДокумента FROM ТипыДокументов WHERE L >= @L AND R <= @R; ";
128  var selectParentSql = "SELECT КодТипаДокумента FROM ТипыДокументов WHERE Parent = @Parent; ";
129 
130  DocTypeParams.ForEach(
131  type =>
132  {
133  var sql = new StringBuilder();
134  switch (type.QueryType)
135  {
136  case DocTypeQueryType.Equals:
137  typeIDs.Add(type.DocTypeID);
138  break;
139 
140  case DocTypeQueryType.WithChildren:
141  sql.Append(@"
142 DECLARE @L int, @R int; ");
143  sql.AppendFormat(@"
144 SELECT @L = L, @R = R FROM ТипыДокументов WHERE КодТипаДокумента = {0}; ", type.DocTypeID);
145  sql.Append(selectLRSql);
146  break;
147 
148  case DocTypeQueryType.WithSynonyms:
149  sql.AppendFormat(@"
150 DECLARE @Parent int;
151 SELECT @Parent = Parent FROM ТипыДокументов WHERE КодТипаДокумента = {0}; ", type.DocTypeID);
152  sql.Append(selectParentSql);
153  break;
154 
155  case DocTypeQueryType.WithChildrenSynonyms:
156  sql.Append(@"
157 DECLARE @L int, @R int; ");
158  sql.AppendFormat(@"
159 SELECT @L = ISNULL(Pr.L, Ch.L), @R = ISNULL(Pr.R, Ch.R)
160 FROM ТипыДокументов Ch LEFT JOIN ТипыДокументов Pr ON Pr.КодТипаДокумента = Ch.Parent WHERE Ch.КодТипаДокумента = {0}; ",
161  type.DocTypeID);
162  sql.Append(selectLRSql);
163  break;
164  }
165 
166  if (type.QueryType != DocTypeQueryType.Equals)
167  {
168  var da = new SqlDataAdapter(sql.ToString(), Config.DS_document);
169  var dt = new DataTable();
170 
171  try
172  {
173  da.Fill(dt);
174  }
175  catch (Exception ex)
176  {
177  Logger.WriteEx(new DetailedException("Ошибка при получении типов документов в DSODoc",
178  ex, da.SelectCommand));
179  throw ex;
180  }
181 
182  for (var i = 0; i < dt.Rows.Count; i++)
183  {
184  var typeID = dt.Rows[i][0].ToString();
185  if (!typeIDs.Contains(typeID)) typeIDs.Add(typeID);
186  }
187  }
188  }
189  );
190  return Convert.Collection2Str(typeIDs);
191  }
192  return string.Empty;
