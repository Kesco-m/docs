183  public DoisserEmployee(int userID)
184  {
185  if (userID == 0) return;
186  Unavailable = false;
187  EmployeeID = userID;
188  Contacts = new List<EmployeeContact>();
189  Positions = new List<EmployeePosition>();
190  WorkPlaces = new List<EmployeeWorkPlace>();
191  CoWorkers = new List<EmployeeCoWorker>();
192  Roles = new List<EmployeeRole>();
193 
194  Replaced = false;
195 
196  Employee = new Employee(userID.ToString());
197 
198  if (Employee == null || Employee.Unavailable)
199  {
200  Unavailable = true;
201  return;
202  }
203  /*Общие сотрудники */
204  if (String.IsNullOrEmpty(Employee.CommonEmployeeID) && Employee.Status == 0 && Employee.PersonEmployeeId == null)
205  {
206  CommonEmployees = GetListCommonEmployeesByIds(Employee.Id);
207  }
208  else
209  {
210  CommonEmployees = new List<Employee>();
211  }
212 
213  EmployeeReplacements = GetEmployeeReplacementById(Employee.Id);
214  EmployeeRepresentatives = GetEmployeeReplacementiveById(Employee.Id);
215  Replaced = EmployeeReplacements.Count != 0;
216  EmployeeStatus = Employee.GetEmployeeStatus(Employee.Status);
217  LastPassage = new EmployeePassage(Employee.Id);
218  IsPersonAdministrator = BelongsToPersonAdministators(Employee.Id);
219 
220  if (Employee.Status != 3)
221  {
222 
223  Contacts = GetEmployeeContacts(Employee.Id);
224 
225  if (Employee.OrganizationId.HasValue)
226  {
227  EmployeeCustomer = new Corporate.PersonCustomer(Employee.Id);
228  }
229 
230  Roles = GetAllEmployeRoles(Employee.Id);
231  Positions = GetEmployeePositions(Employee.Id);
232 
233  WorkPlaces = GetEmployeeWorkPlaces(Employee.Id);
234 
235  WorkPlaces.ForEach(wp =>
236  {
237  wp.CoWorkers = GetEmployeeCoWorkersByWorkPlace(Employee.Id, wp.Id);
238  });
239  }
240 
241  ContactGroups = Contacts
242  .Where(c => c.ContactType == -1)
243  .GroupBy(c => c.ContactType)
244  .Select(g => new ContactTypeGroup
245  {
246  ContactTypeID = g.Key,
247  Items = g.Select(c => new UserInfoContact
248  {
249  Contact = c.Contact,
250  ContactType = c.ContactType,
251  Icon = String.IsNullOrEmpty(c.Icon) ? null : c.Icon,
252  PhoneNumber = (c.ContactType == -1) ? c.Contact : String.Empty,
253  Note = c.Note,
254  @Type = c.ContactTypeDesc,
255  Order = c.Order,
256  InDictionary = c.InDictionary
257  })
258  .OrderBy(c => c.Order).ToList()
259  }).ToList();
260 
261  CanChangeWorkPlace = BelongsToPersonAdministators(Employee.Id);
262  }
