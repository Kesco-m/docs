235  protected override void Render(HtmlTextWriter output)
236  {
237  var currentAsm = Assembly.GetExecutingAssembly();
238  var pagingBarContent =
239  currentAsm.GetManifestResourceStream("Kesco.Lib.Web.Controls.V4.PagingBar.PagingBarContent.htm");
240  if (pagingBarContent == null) return;
241  var reader = new StreamReader(pagingBarContent);
242  var sourceContent = reader.ReadToEnd();
243 
244  sourceContent = sourceContent.Replace(IDTag, ID);
245  sourceContent = sourceContent.Replace(LCurrentPageTag, _resx.GetString("PageBar_lCurrentPage"));
246  sourceContent = sourceContent.Replace(ListenerTag, ListenerPos.ToString(CultureInfo.InvariantCulture));
247  sourceContent = sourceContent.Replace(LTotalRows, _resx.GetString("PageBar_lOf"));
248  sourceContent = sourceContent.Replace(LRowsPerPage, _resx.GetString("PageBar_lRowsPerPage"));
249  sourceContent = sourceContent.Replace(LFirstPage, _resx.GetString("PageBar_lFirstPage"));
250  sourceContent = sourceContent.Replace(LPrevPage, _resx.GetString("PageBar_lPrevPage"));
251  sourceContent = sourceContent.Replace(LNextPage, _resx.GetString("PageBar_lNextPage"));
252  sourceContent = sourceContent.Replace(LLastPage, _resx.GetString("PageBar_lLastPage"));
253  sourceContent = sourceContent.Replace(LTabIndex, TabIndex.ToString(CultureInfo.InvariantCulture));
254 
255  using (TextWriter currentPageTextWriter = new StringWriter())
256  {
257  var currentPageWriter = new HtmlTextWriter(currentPageTextWriter);
258  _currentPageCtrl.RenderControl(currentPageWriter);
259  sourceContent = sourceContent.Replace(CtrlCurNumbTag, currentPageTextWriter.ToString());
260  }
261 
262  using (TextWriter rowsCountTextWriter = new StringWriter())
263  {
264  var rowsCountWriter = new HtmlTextWriter(rowsCountTextWriter);
265  _rowsPerPageCtrl.RenderControl(rowsCountWriter);
266  sourceContent = sourceContent.Replace(CtrlRowPerPageTag, rowsCountTextWriter.ToString());
267  }
268 
269  sourceContent = sourceContent.Replace("\n", "").Replace("\r", "").Replace("\t", "");
270  output.Write(sourceContent);
271  }
